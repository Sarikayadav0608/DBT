mysql> select c.Cname,s.Sname,s.Comm,(o.Amt*s.Comm) as AmtComm
    -> from orders o
    -> join customers c
    -> on c.Cnum=o.Cnum
    -> join salespeople s
    -> on s.Snum=o.Snum
    -> where Rating>100;
